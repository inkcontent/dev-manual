import INKAPI from './inkapi.js'

INKAPI.ready(async () => {

  const EDITOR = INKAPI.editor;

  EDITOR.getText().then(text => {

    console.log(text);

  })

  //Or use async/await

  const text = await EDITOR.getText();
  console.log(text);

});
